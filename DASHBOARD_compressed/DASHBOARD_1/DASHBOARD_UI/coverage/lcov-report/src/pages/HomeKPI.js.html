import React, { useEffect, useState } from 'react';
import { useParams, Link } from "react-router-dom";
import { Icon, Button, Switch, OutlinedInput, Skeleton } from "@mui/material";
import KpiService from '../services/kpiService';
import AssetComponent from '../components/AssetComponent';
import TreeNode from './configpages/KPITreeHome/TreeNode';
import { useKPI, useMenu } from '../hooks';
import { CONTEXT, URL } from '../config';
import Utils from '../utilities';
 
import Banner from "../components/Banner";
import ConfigurationAndSettings from "./configpages/KPITreeHome/ConfigurationAndSettings";
import { permissionAuthorities, permissionFeatures } from '../utilities/AllTables';
 
const HomeKPI = props => {
    const { asset } = useParams();
    const { routes } = useMenu();
    const {toggleTreelist, setToggleTreelist} = useKPI();
    let page = asset;
    const [isLoading, setIsLoading] = useState(false);
    const [listtrees, setListTrees] = useState([]);
    
    const [enableSetting, setEnableSetting] = useState(false);
    const isDomainBased = Utils.soiDomainBased();
    const [searchKPI, setSearchKPI] = useState("");
 
    const toggleSetting = () => {
        setEnableSetting((prev) => !prev);
    };
 
    const handleToggleTreelist = (val) => {
        let uri = !val ? `listPublishedTrees` : `listSavedTrees`;
        setToggleTreelist(val);
        getContent(uri);
    }
 
    const getContent = (uri) => {
        setIsLoading(true);
 
        const assetMenu = routes.filter(item => `assetClassHome` in item);
        if(assetMenu && assetMenu.length){
            let arr = assetMenu[0]['children'].filter(x => x.route === page);
            if(arr.length){
                page = arr[0].title
            }
        }
 
        let group = isDomainBased ? page : null;
        KpiService[uri]({ group })
            .then(res => {
                if (res.data.code === 200) {
                    let data = res.data.response;
                    setListTrees(data)
                } else {
                    setListTrees([])
                }
            })
            .catch(error => console.log(error.message))
            .finally(() =>  setIsLoading(false))
    }
    
    useEffect(() => {
        getContent(!toggleTreelist ? `listPublishedTrees` : `listSavedTrees`);
        setEnableSetting(false);
    }, [props])
 
 
    if(enableSetting){
     return <ConfigurationAndSettings setEnableSetting={setEnableSetting} domain={page}/>
    }
 
    return (
        <div className="container-fluid px-0">
            <Banner />
            {isDomainBased ? <AssetComponent domain={page} /> : null}
            <div className="sub-header">
                <div className={`border-bottom px-3 ${isDomainBased && 'border-top'}`}>
                    <div className="d-flex justify-content-between align-items-center tree-options">
                        <div className='d-flex align-items-center'>
                            <h5 className="font-weight-bold text-primary-blue py-1 my-2 mr-3">{!toggleTreelist ? `Published` : `Drafted`} KPI Tree</h5>
                            <form variant="outlined" style={{ width: "15rem" }}>
                                <OutlinedInput
                                    placeholder="Search by Id/Name"
                                    data-testid="search-input"
                                    value={searchKPI}
                                    onChange={(e) => (e.nativeEvent.inputType) && setSearchKPI(e.target.value)}
                                    style={{ minWidth: "233px", height: "2rem", paddingRight: "8px" }}
                                    endAdornment={ searchKPI !== "" ? (
                                            <Icon className="align-middle text-primary-blue cursor-pointer" title="Clear Search" onClick={() => setSearchKPI("")}>
                                                close
                                            </Icon>
                                        
                                        ):(
                                            <Icon className="align-middle text-primary-blue">
                                                search
                                            </Icon>
                                        )}
                                />
                            </form>
                        </div>
 
                        <div className="controls d-flex align-items-center">
                            {Utils.getPermission(permissionFeatures.KpiTree).includes(permissionAuthorities.Read) && <>
                                <h6 className="m-0 text-uppercase" style={{color: !toggleTreelist ? "#3F88C5" : "#979797"}}>Published</h6>
                                <Switch color="primary" className="mx-2" disabled={isLoading} checked={toggleTreelist} 
                                    onChange={() => handleToggleTreelist(!toggleTreelist)} 
                                    inputProps={{
                                        "aria-label": "controlled",
                                        "data-testid": `toggle__switch_checkbox`,
                                        }}
                                />
                                <h6 className="m-0 mr-2 text-uppercase" style={{ color: toggleTreelist ? "#3F88C5" : "#979797"}}>Drafts</h6>
                                </>
                            }
                            {Utils.getPermission(permissionFeatures.KpiTree).includes(permissionAuthorities.Write) && <>
                                <Button size="small" variant="contained" 
                                    disabled={!toggleTreelist}
                                    data-testid="create-new-tree"
                                    className={`${toggleTreelist && 'bg-primary-blue'} text-white px-3 py-1 mx-3 d-flex`}
                                    component={Link} to={CONTEXT+'analysis/new?domain='+ (page || "Others")}
                                >
                                    <Icon className="text-white fs14 mr-2">add</Icon> Create New KPI Tree
                                </Button>
                            </>}
                            {Utils.getPermission(permissionFeatures.KpiConfiguration).includes(permissionAuthorities.Read) && <>
                                <Button size="small" className="bg-primary-blue text-white px-3 py-1" onClick={() => toggleSetting()}>
                                    <Icon className="text-white fs14 mr-2">settings</Icon> Configuration
                                </Button>
                            </>}
                        </div>
                    </div>
                </div>
                <div className="container-fluid mt-4">
                    <div className="row justify-content-center">
                        {isLoading
                            ? [...Array(9)].map((_, i) => <CardSkeleton key={i} />)
                            : <KpiDisplay toggleTreelist={toggleTreelist} searchKPI={searchKPI} KPIMenuContent={listtrees} />
                        }
                    </div>
                </div>
            </div>
        </div>
    );
}
 
export default HomeKPI;
 
const KpiDisplay = (props) => {
    let savedOrPublished = props.toggleTreelist ? `Saved` : `Published`;
    const publishReadPermission = Utils.getPermission(permissionFeatures.KpiTreePublish).includes(permissionAuthorities.Read);
    const draftReadPermission = Utils.getPermission(permissionFeatures.KpiTree).includes(permissionAuthorities.Read);
    if((props.toggleTreelist && !draftReadPermission) || (!props.toggleTreelist && !publishReadPermission)){
        return (<div className="row justify-content-center">
            <div className="col-8 text-center">
                <img className="p-3" src={`${URL}refresh-session.svg`} alt="refresh-session" />
                <h4 className="text-primary-blue mb-2">Access Denied!</h4>
                <p className="font-weight-normal fs14 mb-3">You do not have sufficient permission to make this request.</p>
            </div>
        </div>)
    }
    const filteredKPITree = props.KPIMenuContent.filter((item) =>
        (props.toggleTreelist ? item?.kpiTreeName.toLowerCase().includes(props.searchKPI.toLowerCase()) : item.treeName.toLowerCase().includes(props.searchKPI.toLowerCase())) ||
        (props.toggleTreelist ? item.kpiTreeId == props.searchKPI : item.treeId == props.searchKPI)
    );
 
    return (
      <>
        {filteredKPITree.length > 0
          ? filteredKPITree.map(item => {
            let treeName = props.toggleTreelist ? item.kpiTreeName : item.treeName;
            let treeId = item.treeId || item.kpiTreeId;
            //let  startDate = item.startDate ? new Date(item.startDate).toLocaleDateString() : '';
            //let  endDate = item.endDate ? new Date(item.endDate).toLocaleDateString() : '';
            return <TreeNode
                    key={treeId}
                    domain={item.kpiTreeDomain}
                    title={item.kpiTreeStatus === 'Modified' ? treeName + "*" : treeName}
                    navLinkTo={CONTEXT + 'analysis/' + treeId}
                    attributes={JSON.parse(item.attributes)}
                    polarity={JSON.parse(item.polarity)}
                    unit={JSON.parse(item.unit)}
                    nodeCount={item.nodeCount}
                    trendLine={item.trendLine}
                    duration={`${item.startDate? Utils.convertShortDate(item.startDate): ''} - ${item.endDate? Utils.convertShortDate(item.endDate) : ''}`}
                    color={ '#ccc'
                        // item.attributes
                        //     ? props.getNodeStatus( item.attributes.attributes.node_status)
                        //     : props.getNodeStatus("")
                    }
                    isDraft={props.toggleTreelist}
                    modifiedDate={item.modifiedDate}
                    refreshDate={item.refreshDate}
                    publishedDate={item.publishedDate}
                />
            })
          : 
            <div className="w-100 row align-items-center justify-content-center my-5 nodata-placeholder">
                <div className="col-md-4">
                <img className="w300" src={`${URL}empty-state.svg`} alt="Empty State" />
                </div>
                <div className="col-md-4 text-center">
                <h4>No {savedOrPublished} KPI Trees!</h4>
                <p className="fs16">There are no {savedOrPublished.toLowerCase()} KPI Trees created yet for this asset. Any new KPI Tree is created in drafts and eventually published once all the needful nodes, cohorts, etc. are created.</p>
                </div>
            </div>
        }
      </>
    );
  };
