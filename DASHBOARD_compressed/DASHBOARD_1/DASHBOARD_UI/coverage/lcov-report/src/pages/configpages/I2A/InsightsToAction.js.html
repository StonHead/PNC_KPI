const InsightsToAction = (props) => {
    const [isNewAction, setIsNewAction] = useState(false);
    const [actionsData, setActionsData] = useState([]);
    const [mode, setMode] = useState(''); // new, edit, delete - Actions
    const [rawData, setRawData] = useState([]);
    const [searchText, setSearchText] = useState('');
    const [isLoading, setIsLoading] = useState(false);
    const [cardView, setCardView] = useState(true);
    const [actionFormData, setActionFormData] = useState({
        actionCohortType: '',
        actionCohortName: '',
        actionName: '',
        actionDesc: '',
        actionStartDate: '',
        actionEndDate: '',
        actionDirectRecommendation: 'N',
 
        actionDurationTypeCode: "1W",
        actionRemarks: "For Testing",
        actionStatus: "NEW",
        actionType: "USER"
    });
    const [allActions, setAllActions] = useState(false);
    const [activeFilter, setactiveFilter] = useState([])
 
    /* for experiments modal */
    const [expMode, setExpMode] = useState('add'); // add, edit, hub -  Experiments
    const [expModal, setExpModal] = useState(false);
    const [expSelected, setExpSelected] = useState(null);
 
 
    const filterMenu = (_record) => {
 
        return (
                <Paper>
 
                    {filterList.map((list, i) => (
                        <MenuList key={i}>
                            <h6 className="font-weight-bold px-3">{list.group}</h6>
                            {
                                list.items.map((item, index) =>
                                    <MenuItem data-testid={`MenuItem-${item.title}`} key={index} onClick={() => onFilterChange(item.value)}>
                                        {
                                            activeFilter.find((emp) => emp[item.value.column] === item.value.value) &&
                                            <Icon className="text-primary-blue mr-2">check</Icon>
                                        }
                                        <ListItemText className="mr-auto">{item.title}</ListItemText>
                                    </MenuItem>
                                )
                            }
                        </MenuList>
                    ))}
 
                </Paper>
        );
    }
 
    const fetchActions = () => {
        setIsLoading(true);
        const dataActions = {
            "actionFilter": JSON.stringify(props.filter),
            "actionKpiTreeId": props.kpiId,
            "actionNodeId": props.nodeId
        }
        listActions(dataActions)
            .then(({ data }) => {
                if (data.code === 200) {
                    setActionsData(data.response);
                    setRawData(data.response)
                    setMode('');
                    setSearchText('');
                } else {
                    setActionsData([]);
                }
            })
            .catch(err => console.log(err))
            .finally(() => setIsLoading(false));
    }
 
    const searchAction = (val) => {
        setactiveFilter([]);
        setSearchText(val);
        let filtered = [...rawData];
        if (val.length > 0) {
            filtered = filtered.filter(item => item.actionName.toLowerCase().includes(val.toLowerCase()));
            setActionsData(filtered)
            return
        }
        setActionsData(filtered)
    }
 
    const onFilterChange = (val) => {
        searchAction('')
        let newArray = [];
        const exists = activeFilter.find((emp) => emp[val.column] === val.value);
        if (!exists) {
            newArray = [...activeFilter, { [val.column]: val.value }];
            setactiveFilter(newArray);
        }
        else {
            const array = [...activeFilter];
            const removeIndex = array.findIndex(item => item[val.column] === val.value);
            array.splice(removeIndex, 1);
            newArray = [...array];
            setactiveFilter(newArray);
        }
 
        filterAccordingtoValue(newArray);
    }
 
    const filterAccordingtoValue = (activeFilterArray) => {
        let filtered = [...rawData];
 
        let myArray = [];
 
        if (activeFilterArray.length > 0) {
            activeFilterArray.map(item => {
                let fArray = filtered.filter(f => f[Object.keys(item)[0]] === item[Object.keys(item)[0]]);
                myArray = [...myArray, ...fArray];
            }
            );
            const uniqueArray = [...new Set(myArray.map(item => item))];
            setActionsData(uniqueArray)
            return
        }
        setActionsData(filtered)
    }
 
    const addAction = (formData) => {

            actionNodeId: props.nodeId
        }
 
        createAction(actionData)

                if (data.code === 201) {

                    fetchActions();

                    window.alert(data.message);
                }
            })
            .catch(err => console.log(err))
 
    }

            actionCohortName: '',

            actionDurationTypeCode: formData.actionDurationTypeCode,
            actionStartDate: formData.actionStartDate,
            actionEndDate: formData.actionEndDate,
            actionId: formData.actionId,
            actionRemarks: formData.actionRemarks,
        }
        updateAction(actionData)
            .then(({ data }) => {
                if (data.code === 200) {
                    setIsNewAction(false);
                    window.alert('Action updated successfully!');
                    fetchActions();
                } else {
                    window.alert(data.message);
                }
            })
            .catch(err => console.log(err))
    }
 
    const handleEditAction = (key) => {
        let record = actionsData[key];
        setIsNewAction(true);
        setActionFormData({
            actionName: record.actionName,
            actionDesc: record.actionDesc,
            actionDurationTypeCode: record.actionDurationTypeCode,
            actionStartDate: record.actionStartDate,
            actionEndDate: record.actionEndDate,
            actionDirectRecommendation: record.actionDirectRecommendation,
            actionId: record.actionId,
            actionFilter: record.actionFilter,
            actionRemarks: record.actionRemarks,
            actionCohortType: record.actionCohortType,
            actionCohortName: record.actionCohortName,
 
        });
        setMode('edit');
    }
 
    const removeAction = (record) => {
        if (!window.confirm("Are you sure you want to delete?")) {
            return false;
        }
 
        deleteAction(record.actionId)
            .then(({ data }) => {
                if (data.code === 200) {
                    window.alert('Action deleted successfully!');
                    fetchActions();
                }
            })
            .catch(err => console.log(err))
    }
 
    const recommendAction = (record) => {
 
        pushAction([{ actionId: record.actionId }])
            .then(({ data }) => {
                if (data.code === 200) {
                    window.alert('Action recommended successfully!');
                    fetchActions();
                }
            })
            .catch(err => console.log(err))
    }
 
    const completeAction = (record) => {
 
        markAsCompleteAction([{ actionId: record.actionId }])
            .then(({ data }) => {
                if (data.code === 200) {
                    window.alert('Action marked as completed!');
                    fetchActions();
                }
            })
            .catch(err => console.log(err))
    }
 
    const cohortPopulation = (cohortName) => {
        let filtered = props.cohortData ? props.cohortData.filter(j => j.segments === cohortName) : [];
        return filtered.length ? filtered[0].noOfAccounts : 0;
    }
 
    const isActionExperiment = () => {
        let actionExp = actionsData.length && actionsData.filter((action) => action.actionExperiments !== null);
        return !actionExp.length;
    }
    useEffect(() => {
        fetchActions();
 
    }, [props.cohortData])
 
    const actionList = () => {
        if(isLoading) {return <ShimmerLoaderActions />}
        if(cardView){
            return (
                actionsData.length > 0 ?
                <div className="row justify-content-center">
                    {actionsData.map((item, i) => (
                        <div className="m-2" key={i}>
                            <CardAction data={item} rawData={rawData} setExpMode={setExpMode} setExpModal={setExpModal} setExpSelected={setExpSelected} fetchActions={fetchActions} attributes={props.attributes} unit={props.unit} cohortPopulation={cohortPopulation} />
                        </div>
                    ))}
                </div>
                :

                window.alert('Fields required!');
                return false;
            }
            props.editAction(formData)
        } else {

                window.alert('Fields required!');
                return false;
            }
            props.addAction(formData)
        }
    };
 
    return (
        <Modal className="action-modal" title="Create New Action" open={props.isNewAction} onOk={handleOk} onCancel={() => props.handleAddAction(false)}>
            <div className="row">
                <div className="col-md-6 mb-3">
                    <FormControl variant="filled" className="w-100">
                        <InputLabel id="demo-simple-select-filled-label">Select Cohort Type</InputLabel>
                        <Select
                            disabled={mode === 'edit' ? true : false}
                            labelId="demo-simple-select-filled-label"
                            id="demo-simple-select-filled"
                            value={formData.actionCohortType}
                            onChange={event => setFormData({ ...formData, actionCohortType: event.target.value })}
                            label="Select Cohort Type" placeholder="Select Cohort Type"
                        >
                            {
                                props.cohortType.map((item, i) => (
                                    <MenuItem key={i} value={item}>{item}</MenuItem>
                                ))
                            }
                        </Select>
                    </FormControl>
                </div>

const experimentCard = (actionsData, mode, expSelected, cohortPopulation, setExpModal) => {
    const { isJsonString } = Utils;
    let cardObj = [];
    if(mode === 'add'){
        if (actionsData.length) {
            actionsData.filter(i => i.actionId === expSelected['actionId']).map(x => {

                let durationInProgress = [];
                if (exp && exp.length) {
                    exp.filter(j => j.expStatus === "IN-PROGRESS").map(z => {
                        durationInProgress.push({
                            expStartDate: z.expStartDate,
                            expEndDate: z.expEndDate
                        })
                    })
                }
 
                cardObj.push({
                    expStatus: 'NEW',
                    expKpiTreeId: x.actionKpiTreeId,
                    expActionId: x.actionId,
                    expName: x.actionName,
                    expDesc: x.actionDesc,
                    actionCohortType: x.actionCohortType,
                    actionCohortName: x.actionCohortName,
                    cohortPopulation: cohortPopulation(x.actionCohortName),
                    sampleMinSize: x.sampleMinSize,
                    primaryMetric: x.primaryMetric,
                    actionCreatedBy: x.actionCreatedBy,
                    actionStartDate: x.actionStartDate,
                    actionEndDate: x.actionEndDate,
                    durationInProgress: durationInProgress
                })
            });
        }
    }
 
    if(mode === 'edit'){
        if (actionsData.length) {
            actionsData.filter(i => i.actionExperiments !== null).map(x => {
                let exp = isJsonString(x.actionExperiments);
                let durationInProgress = [];
                if (exp && exp.length) {
                    exp.filter(k => k.expStatus === "IN-PROGRESS").map(entity => {
                        durationInProgress.push({
                            expStartDate: entity.expStartDate,
                            expEndDate: entity.expEndDate
                        })
                    })
 
                    exp.filter(u => u.expId === expSelected['expId']).map(j => {
                        cardObj.push({
                            expActionId: j.expActionId,
                            expId: j.expId,
                            actionCohortType: x.actionCohortType,
                            actionCohortName: x.actionCohortName,
                            cohortPopulation: cohortPopulation(x.actionCohortName),
                            sampleMinSize: x.sampleMinSize,
                            expName: j.expName,
                            expDesc: j.expDesc,
                            expType: j.expType,
                            actionStartDate: x.actionStartDate,
                            actionEndDate: x.actionEndDate,
                            expStatus: j.expStatus,
                            actionCreatedBy: x.actionCreatedBy,
                            durationInProgress: durationInProgress
                        })
                    })
                }
            })
        }
    }
 
    if(mode === 'hub'){
        if (actionsData.length) {
            actionsData.filter(i => i.actionExperiments !== null).map(x => {
                let exp = isJsonString(x.actionExperiments);
                let durationInProgress = [];
                if (exp && exp.length) {
                    exp.filter(k => k.expStatus === "IN-PROGRESS").map(item => {
                        durationInProgress.push({
                            expStartDate: item.expStartDate,
                            expEndDate: item.expEndDate
                        })
                    })
 
                    exp.filter(u => u.expStatus !== 'NEW').map(j => {
                        cardObj.push({
                            expActionId: j.expActionId,
                            expId: j.expId,
                            actionCohortType: x.actionCohortType,
                            actionCohortName: x.actionCohortName,
                            expName: j.expName,
                            expDesc: j.expDesc,
                            expType: j.expType,
                            actionStartDate: x.actionStartDate,
                            actionEndDate: x.actionEndDate,
                            expStatus: j.expStatus,
                            actionCreatedBy: x.actionCreatedBy,
                            durationInProgress: durationInProgress
                        })
                    })
                }
            })
        }
    }
 
    if(!cardObj.length){
        setExpModal(false);
        window.alert('No Experiments In-progress!');
    }
 
    return cardObj;

function disabledDate(current) {
    // Can not select days before today
    return current < moment().subtract(0, 'day');
}
 
function startActionAlert(startDate) {
    let dt = new Date(startDate);
    if (moment(dt).isValid()) {
        let diff = moment(dt).diff(moment(), 'days');
        if (diff > 0 && diff < 4) {
            return diff;
        }
    }
    return false;
}
