import React, { useEffect, useState } from 'react';
import { Icon, Button, FormControlLabel, FormControl, InputLabel, Select, TextField, RadioGroup, Radio, MenuItem } from '@mui/material';
import { Popover } from "antd";
import { colorRange } from "../../../../utilities/AllTables";
import Utils from "../../../../utilities";
import { URL } from '../../../../config';
import { useConfirmDialog } from '../../../../hooks';
 
const ThresholdSettings = props => {
	const [thresholdType, setThresholdType] = useState("automatic");
	const [threshold, setThreshold] = useState([]);
	const initValue = { filter: {}, threshold: { significant_growth: '', modest_growth: '', no_change: '', modest_loss: '', significant_loss: '' } };
 
	const { treeData, setTreeData } = props;
    const { isConfirmed } = useConfirmDialog();
 
	const handleThresholdType = async (value) => {
		if (value === "automatic") {
			const confirmed = await isConfirmed('Are you sure you want to switch to Automatic, as you can lose the data you have entered.')
			if (confirmed) {
				setTreeData({ ...treeData, colorThreshold: value });
				setThresholdType(value);
			}
		} else {
			setTreeData({ ...treeData, colorThreshold: JSON.stringify([initValue]) });
			setThresholdType(value);
		}
	}
 
	const onChangeTreeData = (name, value, idx, key) => {
		let parsed = Utils.isJsonString(treeData.colorThreshold);
		parsed[idx] = parsed[idx]['threshold'].hasOwnProperty(`significant_growth`) ? { ...parsed[idx], [key]: {...parsed[idx][key], [name]: value} } : initValue;
		setTreeData({ ...treeData, colorThreshold: JSON.stringify(parsed) });
	}
 
	const colorThreshold = () => {
		let arr = treeData?.colorThreshold !== "automatic" ? treeData && Utils.isJsonString(treeData?.colorThreshold) : [];
		let item = !((arr[0] && arr[0]['threshold']) || {}).hasOwnProperty(`significant_growth`);
 
		let mapped = colorRange.map(val => ({ key: val.val, label: val.label, range: '', color: val.color }));
 
		if (treeData?.colorThreshold === "automatic" || item) {
			return mapped;
		}
 
		if(Array.isArray(arr)){
			mapped = Object.entries(arr[0]['threshold']).map(([k, v]) => {
				let col = colorRange.filter(prop => prop.val === k);
				if(col.length)
					return { key: col[0].val, label: col[0].label, range: v, color: col[0].color }
				else
					return []
			}
			);
		}
 
		return mapped;
	}
 
	const addThreshold = value => {
		let parsed = Utils.isJsonString(treeData.colorThreshold);
		let filter = {};
		treeData.chosenFilter.forEach(item => { filter[item.label] = ''; });
		
		setTreeData({ ...treeData, colorThreshold: JSON.stringify([...parsed, {...value, filter}]) });
	}
 
	const removeThreshold = idx => {
		let parsed = Utils.isJsonString(treeData.colorThreshold);
		parsed.splice(idx, 1);
		setTreeData({ ...treeData, colorThreshold: JSON.stringify(parsed) });
	}
 
	useEffect(() => {
		if(treeData?.colorThreshold !== "automatic"){
			let arr = Utils.isJsonString(treeData.colorThreshold);
			if(Array.isArray(arr)){
				setThresholdType("manual");
			}
		}
		setThreshold(colorThreshold());
	}, [])
 
	return (
		<div className="row">
			<div className="col-md-12">
				<div className="d-flex align-items-center">
					<RadioGroup className="d-flex flex-row px-1" value={thresholdType} onChange={e => handleThresholdType(e.target.value)} aria-label="position">
						<FormControlLabel className="mr-5" value="automatic" control={<Radio color="primary" />}
							label={
								<span>
									<h6 className="fs14 font-weight-bold m-0">Automatic</h6>
									<p className="small m-0 d-none">(Recommended)</p>
								</span>
							}
						/>
						<FormControlLabel value="manual" control={<Radio color="primary" />}
							label={
								<span>
									<h6 className="fs14 font-weight-bold m-0">Set Manually</h6>
									<p className="small m-0 d-none">For Advanced users</p>
								</span>
							}
						/>
					</RadioGroup>
					<Popover arrow placement="rightTop"
						overlayStyle={{ maxWidth: "500px", width: "100%" }}
						content={<img className="img-fluid p-1" src={`${URL}thresholds-settings.png`} alt="Example Analysis" />}>
						<Icon fontSize="medium" className="text-primary ml-3 mb-2">info_outlined</Icon>
					</Popover>
				</div>
				{thresholdType === "manual" &&
					<>
					{
						Utils.isJsonString(treeData.colorThreshold).map((thres, idx) =>
							<div key={idx} className="mt-3">
									<div className="row">
										<div className="col-md-12 d-flex align-items-center mb-3">
											{/* <h6 className="mb-0 ml-1">Threshold <span className="small bold500 text-primary-blue">- Default</span></h6> */}
											<h6 className="mb-0 ml-1 mr-4">Threshold {idx+1} <span className="small bold500 text-primary-blue">- {idx === 0 ? `Default` : `Exception`}</span></h6>
												{idx !== 0 &&
													<Button size="small" color="primary" data-testid="delete-btn" onClick={() => removeThreshold(idx)}>
														<Icon>delete_forever</Icon>
													</Button>
												}
										</div>
 
										<div className="col-md-12 row ml-auto">
											{idx !== 0 && 
												treeData.chosenFilter
												// .filter((fil) => fil.name !== '')
												.map((filterRow, filterIndex) => (
													<div key={filterIndex} className="col-md-4 pl-2 mb-4">
														<FormControl size="small" variant="outlined" className="w-100">
															<InputLabel htmlFor="outlined-age-native-simple">{filterRow.name}</InputLabel>
															<Select
																labelId="demo-simple-select-outlined-label"
																id="demo-simple-select-outlined"
																label={filterRow.name}
																value={`filter` in thres ? thres.filter[filterRow.name] : ''}
																onChange={e => onChangeTreeData(filterRow.name, e.target.value, idx, `filter`)}
															>
																{filterRow.values.map((val, valInd) => (
																	<MenuItem key={valInd} value={val}>{val}</MenuItem>
																))}
															</Select>
														</FormControl>
													</div>
												))}
										</div>
 
										{threshold?.map((item, i) => (
											<div key={i} className="col-md-4 d-flex align-items-center mb-4">
												<Icon style={{ color: item.color }}>stop</Icon>
												<TextField label={item.label} size="small" variant="outlined" autoComplete="soff" className="w-100"
													value={`threshold` in thres ? thres.threshold[item.key] : ''}
													onChange={e => onChangeTreeData(item.key, e.target.value, idx, `threshold`)}
												/>
											</div>
										))}
									</div>
							</div>
						)
					}
					{
						(treeData.chosenFilter.some(x => x.hasOwnProperty('name') && x.name.length)) 
						&& (Utils.isJsonString(treeData.colorThreshold).length < 5)
						&& <Button color="primary" size="small" onClick={() => addThreshold(initValue)}>Add Exception Threshold</Button>
					}
					</>
				}
			</div>
		</div>
	);
}
 
export default ThresholdSettings;
