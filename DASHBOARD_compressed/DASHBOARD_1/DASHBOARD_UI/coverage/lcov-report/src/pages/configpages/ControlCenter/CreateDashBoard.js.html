import { useEffect, useState } from 'react';
 
import { Modal } from 'antd';
import { Box, Button, Tab } from '@mui/material';
import { TabContext, TabList, TabPanel } from '@mui/lab';
 
import { ReactComponent as HandSVG } from './Images/hand.svg';
import { ReactComponent as InstantSVG } from './Images/instant.svg';
import { ReactComponent as GraphSVG } from './Images/graph.svg';
 
import { useSnackBar } from "../../../hooks";
 
import './CreateDashBoard.scss';
 
import Settings from './Creation/CCSettings/Settings';
import SelectInsights from './Creation/SelectInsights/SelectInsights';
// import DesignControlCenter from './Creation/DesignControlCenter';
import AnalystReview from './Creation/AnalystReview';
 
function isObjectNotEmpty(obj) {
  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      const value = obj[key];
      if (value === '' || (Array.isArray(value) && value.length === 0)) {
        return false;
      }
    }
  }
  return true;
}
 
const CreateDashBoard = ({
  open,
  handleClose,
  disabledTabs,
  enableSettingsTab,
  enableInsightsTab,
  enablePreviewTab,
  enableAnalystTab
}) => {
  const [tabSelect, setTabSelect] = useState(1)
  const tabsList = ['Info Settings', 'Select Insights', 'Design Control Center', 'Analyst Review'];
  const tabTitles = ['Select Data', 'Design Control Center', 'Insights Prioritization', 'Analyst Review'];
  const [tabsValidated, setTabsValidated] = useState([0, 0, 0, 0])
 
  const [controlCenterSettingsData, setControlCenterSettingsData] = useState({
    controlCenterName: "",
    selectedExecutiveUser: [],
    selectedKpiTrees: []
  });  //For Settings Tab
 
  const [selectedRowKeys, setSelectedRowKeys] = useState([]); //For SelectInsights Tab
 
  useEffect(() => {
    validateSettingsTab()
  }, [tabSelect, controlCenterSettingsData.controlCenterName, controlCenterSettingsData.selectedExecutiveUser, controlCenterSettingsData.selectedKpiTrees])
 
  useEffect(() => {
    validateInsightsTab()
  }, [tabSelect, selectedRowKeys])
 
  useEffect(() => {
    validateAnalystTab()
  }, [tabSelect, selectedRowKeys])
 
  const { snackBar } = useSnackBar()
 
  const validateSettingsTab = () => {
    if (isObjectNotEmpty(controlCenterSettingsData)) {
      enableInsightsTab();
      setTabsValidated([1, 0, 0, 0])
    } else {
      enableSettingsTab();
      setTabsValidated([0, 0, 0, 0])
    }
  };
 
  const validateInsightsTab = () => {
    if (selectedRowKeys.length) {
      enablePreviewTab();
      enableAnalystTab();
      setTabsValidated([1, 1, 1, 0])
    } else {
      enableInsightsTab();
      setTabsValidated([1, 1, 0, 0])
    }
  };
 
  const validateAnalystTab = () => {
    if (selectedRowKeys.length) {
      setTabsValidated([1, 1, 1, 1])
    } else {
      enablePreviewTab();
      setTabsValidated([1, 1, 1, 0])
    }
  };
 
  const setControlCenterSettingsDataHandler = (key, value) => {
    let data = {
      ...controlCenterSettingsData,
      [key]: value
    }
    setControlCenterSettingsData(data)
  }
 
  const goBackHandler = () => {
    tabSelect > 1 && setTabSelect((prev) => --prev)
  }
 
  const validateSettingsNext = () => {
    if (!controlCenterSettingsData.controlCenterName.trim()) {
      snackBar({ status: true, severity: "error", message: "Please Enter a Name for the Control Center!" });
      return false;
    }
 
    if (!controlCenterSettingsData.selectedExecutiveUser.length == 0) {
      snackBar({ status: true, severity: "error", message: "Please Select atleast 1 Executive User!" });
      return false;
    }
 
    if (controlCenterSettingsData.selectedKpiTrees.length === 0) {
      snackBar({ status: true, severity: "error", message: "Please Select atleast 1 KPI Tree!" });
      return false;
    }
 
    return true;
  }
 
  const validateInsightsNext = () => {
    if (!selectedRowKeys.length) {
      snackBar({ status: true, severity: "error", message: "Please Select atleast 1 Insight!" });
      return false;
    }
 
    return true
  }
 
  const validatePreviewNext = () => {
    return true
  }
 
  const validateAnalystPublish = () => {
    return true
  }
 
  const validateNext = (tab) => {
    if (tab === 1) return validateSettingsNext();
    else if (tab === 2) return validateInsightsNext();
    else if (tab === 3) return validatePreviewNext()
    else if (tab === 4) return validateAnalystPublish()
  }
 
  const goNextHandler = () => {
    if (!validateNext(tabSelect)) return;
 
    tabSelect === 1 && enableInsightsTab()
    tabSelect === 2 && enablePreviewTab()
    tabSelect === 3 && enableAnalystTab()
    tabSelect < 4 && setTabSelect((prev) => ++prev)
  }
 
  const publishHandler = () => {
    handleClose();
  }
 
  const handleChange = (_event, newValue) => {
    setTabSelect(newValue);
  };
 
  return (
    <div>
      <Modal
        open={open}
        width={1300}
        style={{ top: 40 }}
        onCancel={handleClose}
        className={'create-modification'}
        closable={false}
        footer={null}
        title={
          <div className='d-flex'>
            <div className='d-flex title-name'>
              <div className='font-weight-bold text-primary-blue title-row mb-1'>
                Configure Control Center
              </div>
              <div className='bar-line mx-4'></div>
              <div className='sub-title-row'>{tabTitles[tabSelect - 1]}</div>
            </div>
          </div>
        }
      >
        {/* Tabs List */}
        <div>
          <Box className="position-absolute w-100"
            sx={{ position: 'fixed', bottom: '-50px', left: '0px', bgcolor: '#c1c7d0', borderRadius: '8px', zIndex: 9 }}>
            <div className='bg-white pb-4 d-flex justify-content-center'>
              <Button className='text-primary-blue text-capitalize mx-2' onClick={goBackHandler}>Back</Button>
              {tabSelect < 4 ?
                <Button className='text-white bg-primary-blue text-capitalize' onClick={goNextHandler}>Next</Button>
                : <Button className='text-white bg-primary-blue text-capitalize' onClick={publishHandler}>Publish</Button>
              }
            </div>
            <TabContext value={tabSelect}>
              <TabList
                onChange={handleChange}
                className="br8 tab-theme-1"
                textColor="primary"
                indicatorColor="primary"
                centered
                variant="fullWidth"
              >
                <Tab value={1} disabled={disabledTabs.settingsTab} className={`no-outline text-capitalize ${tabSelect === 1 && 'bg-white'} ${tabsValidated[0] && tabSelect > 1 && 'bg-grey1'}`} icon={<><InstantSVG /><span className='text-width px-2'>{tabsList[0]}</span></>} iconposition="start" />
                <Tab value={2} disabled={disabledTabs.insightsTab} className={`no-outline text-capitalize ${tabSelect === 2 && 'bg-white'} ${tabsValidated[1] && tabSelect > 2 && 'bg-grey1'}`} icon={<><HandSVG /><span className='text-width px-2'>{tabsList[1]}</span></>} iconposition="start" />
                <Tab value={3} disabled={disabledTabs.previewTab} className={`no-outline text-capitalize ${tabSelect === 3 && 'bg-white'} ${tabsValidated[2] && tabSelect > 3 && 'bg-grey1'}`} icon={<><GraphSVG /><span className='text-width px-2'>{tabsList[2]}</span></>} iconposition="start" />
                <Tab value={4} disabled={disabledTabs.analystReview} className={`no-outline text-capitalize ${tabSelect === 4 && 'bg-white'}`} icon={<><GraphSVG /><span className='text-width px-2'>{tabsList[3]}</span></>} iconposition="start" />
              </TabList>
            </TabContext>
          </Box>
        </div>
        <TabContext value={tabSelect} className="tab-context">
          <TabPanel value={1}>
            <Settings
              controlCenterSettingsData={controlCenterSettingsData}
              setControlCenterSettingsData={setControlCenterSettingsData}
              setControlCenterSettingsDataHandler={setControlCenterSettingsDataHandler}
            />
          </TabPanel>
          <TabPanel value={2}>
            <SelectInsights
              selectedRowKeys={selectedRowKeys}
              setSelectedRowKeys={setSelectedRowKeys}
            />
          </TabPanel>
          <TabPanel value={3}>
            {/* <DesignControlCenter /> */}
          </TabPanel>
          <TabPanel value={4}>
            <AnalystReview />
          </TabPanel>
        </TabContext>
      </Modal>
    </div>
  );
};
 
export default CreateDashBoard;
