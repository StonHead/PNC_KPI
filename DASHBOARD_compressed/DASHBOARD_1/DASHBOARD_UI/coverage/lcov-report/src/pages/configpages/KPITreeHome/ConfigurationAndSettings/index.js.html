import React, { useState } from "react";
import { Tab, Button } from "@mui/material";
import { TabContext, TabList, TabPanel } from "@mui/lab";
import { permissionAuthorities, permissionFeatures, settingTabs } from "../../../../utilities/AllTables";
import GeneralManagement from "./Tabs/GeneralManagement";
import MetricsManagement from "./Tabs/MetricsManagement";
import CohortManagement from "./Tabs/CohortManagement";
import DataCatalogManagement from "./Tabs/DataCatalogManagement";
import ParameterManagement from "./Tabs/ParameterManagement";
import Utils from "../../../../utilities";
 
const ConfigurationAndSettings = (props) => {
  const [value, setValue] = useState("parameter-management");
 
  const handleChange = (_event, newValue) => {
    setValue(newValue);
  };
 
  const {businessday, metric, cohort, dataCatalog, parameter} = settingTabs;
 
  return (
    <div className="mb-5">
      <TabContext value={value} className="tab-context">
        <div className="border-bottom px-3">
          <div className="d-flex align-items-center justify-content-between">
            <h5 className="font-weight-bold text-primary-blue py-1 my-2">Configuration</h5>
            <Button className="bg-primary-blue box-shadow-none ml-2" color="primary" size="small" variant="contained" onClick={() => props.setEnableSetting(false)}>Go Back</Button>
          </div>
        </div>
        <div className="config-tabs d-flex justify-content-center">
          <TabList
            onChange={handleChange}
            className="tab-list"
            indicatorColor="primary"
            textColor="primary"
          >
            {/* {Utils.getPermission(permissionFeatures.BusinessDay).includes(permissionAuthorities.Read) &&
              <Tab className="no-outline" data-testid={`${businessday.label}-${businessday.id}`} key={businessday.id} label={businessday.label} value={businessday.value} />
            } */}
            {Utils.getPermission(permissionFeatures.DataCatalogManagement).includes(permissionAuthorities.Read) &&
              <Tab className="no-outline" data-testid={`${dataCatalog.label}-${dataCatalog.id}`} key={dataCatalog.id} label={dataCatalog.label} value={dataCatalog.value} />
            }
            {Utils.getPermission(permissionFeatures.ParameterManagement).includes(permissionAuthorities.Read) &&
              <Tab className="no-outline" data-testid={`${parameter.label}-${parameter.id}`} key={parameter.id} label={parameter.label} value={parameter.value} />
            }
            {Utils.getPermission(permissionFeatures.MetricManagement).includes(permissionAuthorities.Read) &&
              <Tab className="no-outline" data-testid={`${metric.label}-${metric.id}`} key={metric.id} label={metric.label} value={metric.value} />
            }
            {Utils.getPermission(permissionFeatures.CohortManagement).includes(permissionAuthorities.Read) &&
              <Tab className="no-outline" data-testid={`${cohort.label}-${cohort.id}`} key={cohort.id} label={cohort.label} value={cohort.value} />
            }
          </TabList>
        </div>
      </TabContext>
 
      <TabContext value={value} className="tab-context">
            {/* {Utils.getPermission(permissionFeatures.BusinessDay).includes(permissionAuthorities.Read) &&
              <TabPanel className="container" value={businessday.value}>
                <GeneralManagement />
              </TabPanel>
            } */}
            {Utils.getPermission(permissionFeatures.MetricManagement).includes(permissionAuthorities.Read) &&
              <TabPanel className="container" value={metric.value}>
                <MetricsManagement />
              </TabPanel>
            }
            {Utils.getPermission(permissionFeatures.CohortManagement).includes(permissionAuthorities.Read) &&
              <TabPanel className="container" value={cohort.value}>
                <CohortManagement />
              </TabPanel>
            }
            {Utils.getPermission(permissionFeatures.DataCatalogManagement).includes(permissionAuthorities.Read) &&
              <TabPanel className="container" value={dataCatalog.value}>
                <DataCatalogManagement />
              </TabPanel>
            }
            {Utils.getPermission(permissionFeatures.ParameterManagement).includes(permissionAuthorities.Read) &&
              <TabPanel className="container" value={parameter.value}>
                <ParameterManagement />
              </TabPanel>
            }
      </TabContext>
    </div>
  );
};
 
export default ConfigurationAndSettings;
