import Tree from "react-d3-tree";

export const NodeLabel = props => {
  const { nodeDatum, toggleNode, foreignObjectProps, nHeight, drawerOpen, unit, currentNode, orientation, isEdit, options, setOptions, showDrawer, addChild, removeChild } = props;
 
  const getNodeStatus = (status) => {
    let selectedThreshold = colorRange.filter((col) => col.val === status);
    if (selectedThreshold.length > 0) {
      return selectedThreshold[0].color;
    }
    return defaultColor;
  };
 
  const getMetrics = () => {
    const { attributes } = nodeDatum;
    if(attributes && attributes.constructor === Object){
      return Object.entries(attributes)
        .filter(([key]) => key !== "node_status");
    }else if(attributes && attributes.constructor === Array){
      return attributes
        .filter(metr => metr.show)
        .flatMap(metric => Object.entries(metric)
          .filter(([key]) => key !== "show")
        )
    }else{
      return null;
    }
  }
 
  return (
      <foreignObject {...foreignObjectProps}>
      {/* <div className="insights-label">17 Insights</div> */}
      <div className="tree-node position-relative" style={{ border: currentNode === nodeDatum.nodeId ? "2px solid #365C84" : (isEdit && nodeDatum.error !== "") ? "2px solid #ff4141" : "1px solid #9BADC1"}}>
        <div className={`node-data ${orientation && 'pb-3'} ${isEdit && 'pr-4'}`} onClick={() => nodeDatum.childrenIdList && drawerOpen(nodeDatum)}>
          <div title={nodeDatum.name} className="node-title mb-1">{nodeDatum.name !== "" ? nodeDatum.name : "No Title"}</div>
          { isEdit ?
            (nodeDatum.attributes && (nodeDatum.attributes.constructor === Object))
            ? Object.entries(nodeDatum.attributes).map((key, i) => {
                  return <div key={i} className="d-flex flex-row align-items-center justify-content-between px-1">
                    <span>{key[1]}</span>
                    <span>{key[0]}</span>
                  </div>
                })
            : (nodeDatum.attributes && nodeDatum.attributes.constructor === Array)
            ? nodeDatum.attributes?.map((metric) => {
                return Object.entries(metric).map((key, idx) => {
                  
                  return (key[0] !== "show") && (
                    <div key={idx} className="d-flex flex-row align-items-center justify-content-between px-1">
                      {key[1] !== "undefined" && key[0] !== "undefined" && (
                        <>
                          <span className="node-attr-2" title={key[0]}>{`${Utils.trunc(key[0], 11)}`} <span className="font-weight-bold">{`(${key[1].slice(3)})`}</span></span>
                          <div className="text-nowrap">
                            <span className="node-attr-3">{key[1].slice(0, 3)}</span>
                            <Icon className="align-middle ml-1" style={{ color: (metric.show || !metric.hasOwnProperty("show")) ? "green" : "red", fontSize: "1rem" }}>check_circle_sharp</Icon>
                          </div>
                        </>
                      )}
                    </div>
                  );
                });
              })
            : null
            
            :
 
            getMetrics()
              .map(([key, value]) => {
                // const isDollarOrPound = unit && (unit[key] === "$" || unit[key] === "#");
                // const formattedValue = isDollarOrPound ? Utils.shortInt(value) : value.toLocaleString("en-US");
                const unitValue = unit ?(unit[key] ? ` (${unit[key]})` : '' ): null;
 
                return (
                  <div key={key} className="d-flex flex-row align-items-center justify-content-between pl-1 pr-2">
                    <div><span className="node-attr-2" title={key}>{Utils.trunc(key, 11)}</span><span className="font-weight-bold">{unitValue ?? ''}</span></div>
                    <div>
                      <span className="node-attr-3" title={parseFloat(value).toLocaleString()}>{Utils.shortInt(value)}</span>
                    </div>
                  </div>
                );
              })
          }
        </div>
 
 
        {
          isEdit
            ? (Utils.getPermission(permissionFeatures.KpiTree).includes(permissionAuthorities.Write) && <EditOptions nodeData={nodeDatum} options={options} setOptions={setOptions} showDrawer={showDrawer} addChild={addChild} removeChild={removeChild} />)
            : <div className={`position-absolute node-status ${!orientation ? "node-status-horizontal" : "node-status-vertical"}`}
              style={{ backgroundColor: getNodeStatus(nodeDatum.attributes && nodeDatum.attributes.node_status) }}>
            </div>
        }
 
        {nodeDatum.children.length ? (
          <>
            <div className={`row justify-content-center align-items-center node-action ${!orientation ? "node-action-horizontal" : "node-action-vertical"}`}
              style={{ top: ((nHeight / 2) - 14) + 'px' }}
              onClick={toggleNode}
            >
              <Icon color="inherit">
                {nodeDatum.children.length && nodeDatum.__rd3t.collapsed ? 'keyboard_double_arrow_right' : 'keyboard_double_arrow_left'}
              </Icon>
            </div>
 
            {!isEdit && nodeDatum.relation.length ?
              <Popover
                placement={orientation ? 'right' : 'top'}
                content={
                  <div className="text-primary-blue small bold500 rounded py-1 px-2">{nodeDatum.relation}</div>
                }
              >
                <div className={`text-center ${nodeDatum.__rd3t.collapsed && 'invisible'} ${!orientation ? "node-function-horizontal" : "node-function-vertical"}`}
                  style={!orientation ? { top: ((nHeight / 2) - 16) + 'px' } : { bottom: (-((nHeight / 2) + 16)) + 'px' }}
                >
                  <p className="font-italic text-primary-blue40">fx</p>
                </div>
              </Popover>
              : null
            }
          </>
        ) : null}
      </div>
      </foreignObject>
  )
}
 
class TreeCalculatedChart extends React.Component {
  constructor(props) {
    super(props);
 
    this.state = {
      zoom: 0.7,
      direction: "horizontal",
      nodeDataLength: 0,
      nHeight: 100,
      tRect:{},
      options: '',
    };
 
    this.t = undefined;
    this.start = 100;
 
    this.onMouseDown = this.onMouseDown.bind(this);
    this.onMouseUp = this.onMouseUp.bind(this);
    
    this.repeat = this.repeat.bind(this);
    this.zoom = this.zoom.bind(this);
    this.zoomOut = this.zoomOut.bind(this);
    this.zoomOutRepeat = this.zoomOutRepeat.bind(this);
    this.zoomOutDown = this.zoomOutDown.bind(this);
 
  }
 
  onMouseDown() {
    this.repeat();
  }
 
  repeat() {
    this.zoom();
    this.t = setTimeout(this.repeat, this.start);
    this.start = this.start / 2;
  }
 
  zoom() {
    if (this.state.zoom < 1) {
      this.setState({ zoom: this.state.zoom + 0.1 });
    }
  }
 
  zoomOut() {
    if (this.state.zoom > 0.3) {
      this.setState({ zoom: this.state.zoom - 0.1 });
    }
  }
 
  zoomOutRepeat() {
    this.zoomOut();
    this.t = setTimeout(this.zoomOutRepeat, this.start);
    this.start = this.start / 2;
  }
 
  zoomOutDown(e) {
    e.preventDefault();
    this.zoomOutRepeat();
  }
 
  onMouseUp() {
    clearTimeout(this.t);
    this.start = 100;
  }
 
  setNodeDataLength = (length) => {
    this.setState({ nodeDataLength: length })
  }
 
  setOptions = id => {
    this.setState({ options: id })
  }
 
  colorThreshold = () => {
    let arr = this.props.colorThreshold !== "automatic" ? Utils.isJsonString(this.props.colorThreshold) : [];
		let item = !((arr[0] && arr[0]['threshold']) || {}).hasOwnProperty(`significant_growth`);
    
    let mapped = colorRange.map(val => ({ key: val.val, label: val.label, range: '', color: val.color }));
    
    if(this.props.colorThreshold === "automatic" || item){
      return mapped;
    }
 
    if(Array.isArray(arr)){
      let filters = this.props.treeFilter.reduce((obj, {name, selected}) => ({...obj, [name]: selected}), {});
      let index = arr.findIndex(item => JSON.stringify(item.filter) === JSON.stringify(filters));
      let idx = (index >= 0) ? index : 0;
 
      mapped = Object.entries(arr[idx]['threshold']).map(([k, v]) => {
        let col = colorRange.filter(prop => prop.val === k);
        if(col.length)
          return { key: col[0].val, label: col[0].label, range: v, color: col[0].color }
        else
          return []
        }
      );
    }
 
    return mapped;
 
  }
 
  componentDidMount = () => {
    let tRect = document.querySelector('.rd3t-tree-container')?.getBoundingClientRect();
    let nHeight = document.querySelector('.tree-node')?.offsetHeight;
    this.setState({ nHeight, tRect });
  }
 
  render() {
    const { drawerOpen, unit, polarity, currentNode, orientation, isEdit, showTree, addChild, removeChild, showDrawer } = this.props;
    const { nHeight } = this.state;
    const setNodeDataLength = this.setNodeDataLength;
 
    return (
      <>
        <div id="treeWrapper" className="zoomControl" style={{ background: this.props.isDarkMode ? `url('${URL}dot-bg-dark.svg'), #060606` : `url('${URL}dot-bg.svg'), #fafbfc` }}>
          {showTree ?
            <Tree
              collapsible={true}
              data={this.props.treeData[0]}
              orientation={this.props.orientation ? "vertical" : "horizontal"}
              zoom={this.state.zoom}
              pathFunc="step"
              depthFactor={isEdit ? (nHeight * 2) : 0}
              // initialDepth={2}
              // allowForeignObjects
              nodeSize={this.props.orientation ? { x: isEdit ? 110 : nHeight, y: isEdit ? 200 : (nHeight * 2) } : { x: 400, y: nHeight / 2 }}
              separation={{ siblings: 3.5, nonSiblings: 3.5 }}
              enableLegacyTransitions={true}
              renderCustomNodeElement={(rd3tProps) =>
                NodeLabel({
                  ...rd3tProps,
                  foreignObjectProps: {
                    width: 250,
                    height: 1,
                    x: -120,
                    y: -(nHeight / 2),
                  },
                  nHeight,
                  drawerOpen,
                  unit,
                  polarity,
                  currentNode,
                  orientation,
                  isEdit,
                  addChild,
                  removeChild,
                  showDrawer,
                  options: this.state.options,
                  setOptions: this.setOptions,
                  setNodeDataLength
                })
              }
              translate={{
                x: this.state.tRect?.width / 2,
                y: window.innerHeight / 6,
              }}
            />
            : null}
        </div>
 
        <div className="tree-legend tree-legend-right" style={{ width: "2.5rem", right: "2rem", top: "2rem" }}>
          <div className="d-flex flex-column justify-content-center bg-white border zoom-links">
            <button className="zoomIn py-2 border-bottom bg-white text-primary-blue"
              onMouseUp={this.onMouseUp} onMouseDown={this.onMouseDown} data-testid="add-btn">
              <Icon fontSize="small" className="align-middle font-weight-bold">add</Icon>
            </button>
            <button className="zoomOut py-2 bg-white text-primary-blue"
              onMouseUp={this.onMouseUp} onMouseDown={this.zoomOutDown} data-testid="remove-btn">
              <Icon fontSize="small" className="align-middle font-weight-bold">remove</Icon>
            </button>
          </div>
        </div>
 
        {this.props.colorThreshold &&
          <div className="tree-legend tree-legend-left">
            <ul className="d-flex flex-column align-items-start justify-content-center pl-4">
              {this.colorThreshold().map((item, i) =>
                <li key={i} title={item.range} className="d-flex flex-row align-items-center justify-content-center mb-2">
                  <div className={`color-block ${item.range.length ? 'color-block-enhanced' : ''}`} style={{ backgroundColor: item.color }}></div>
                  <div className="legend-range">{item.label}<br /><span className="text-dark">{convertValues(item.range)}</span></div>
                </li>
              )}
            </ul>
          </div>
        }
 
      </>
    );
  }
}
 
export const EditOptions = ({ nodeData, options, setOptions, addChild, removeChild, showDrawer }) => {
  
  return (
    <div className="d-flex flex-column justify-content-center position-absolute edit-options">
      {options !== nodeData.nodeId ?
        <IconButton data-testid="more-btn" className="outline-none p-1" size="small" color="primary" aria-label="edit" title="Open Options"
          onClick={() => setOptions(nodeData.nodeId)}>
          <Icon className="font-weight-bold">more_vert</Icon>
        </IconButton>
        :
        <IconButton data-testid="close-btn" className="outline-none p-1" size="small" color="primary" aria-label="close" title="Close Options"
          onClick={() => setOptions('')}>
          <Icon className="font-weight-bold">close</Icon>
        </IconButton>
      }
 
      <div className="position-absolute edit-options"
        style={options === nodeData.nodeId ? { left: '70px', visibility: 'visible', opacity: 1, transition: 'all 0.2s linear 0s' } : { left: '50px', visibility: 'hidden', opacity: 0, transition: 'all 0.2s linear 0s' }}>
        <IconButton data-testid="edit-btn" className="outline-none bg-white p-2" size="small" color="primary" aria-label="Edit Node details" title="Edit Node details"
          onClick={() => showDrawer(nodeData)} style={{ margin: '0 0 20px -20px' }}>
          <Icon className="font-weight-bold">drive_file_rename_outline</Icon>
        </IconButton>
        <IconButton data-testid="add-btn" className="outline-none bg-white p-2" size="small" color="primary" aria-label="Add new Node" title="Add new Node"
          onClick={() => addChild(nodeData)} style={{ margin: '0 0 20px 0' }}>
          <Icon className="font-weight-bold">add_circle_outline</Icon>
        </IconButton>
        <IconButton data-testid="copy-btn" className="d-none outline-none bg-white p-2" size="small" color="primary" aria-label="Clone Node" title="Clone Node"
          onClick={() => addChild(nodeData)} style={{ margin: '0 0 20px 0' }}>
          <Icon className="font-weight-bold">content_copy</Icon>
        </IconButton>
        {nodeData.nodeId !== '1' &&
          <IconButton data-testid="delete-btn" className="outline-none bg-white p-2" size="small" color="primary" aria-label="Delete Node" title="Delete Node"
            onClick={() => removeChild(nodeData.nodeId)} style={{ margin: '0 0 0 -20px' }}>
            <Icon className="font-weight-bold">delete_outline</Icon>
          </IconButton>
        }
      </div>
    </div>
  )
}
 
function convertValues(str){
  let pattern = /(\d+(\.\d+)?)/g;
  let intValues = str.match(pattern);
  let convertedValues = [];
 
  if(intValues){
    convertedValues = intValues.map(val => Utils.shortInt(Number(val)));
  }
 
  return str.replace(pattern, () => convertedValues.shift());
}
