import React, { useEffect, useState } from 'react';
import { FormControl, FormControlLabel ,FormHelperText, Icon, InputLabel, Select, MenuItem, Switch } from '@mui/material';
import { Popover } from "antd";
 
import { URL } from '../../../../config';
import kpiService from "../../../../services/kpiService";
import cohortPreviewExportService from '../../../../services/cohortPreviewExportService';
import SQLEditor from "../../../../utilities/SQLEditor";
 
const DataConfigurations = props => {
	const [connectionsList, setConnectionsList] = useState([]);
	const [enableCohortOps, setEnableCohortOps] = useState(false);
	const [isFocused, setFocused] = useState(false);
	const { treeData, setTreeData } = props;
 
	const retrieveConnectionsList = () => {
		kpiService.getAllConnections()
			.then(({ data }) => {
				if (data.code === 200) {
					let selectedConnections = data.response.map((connection) => connection);
					setConnectionsList([...selectedConnections]);
				} else {
					setConnectionsList([]);
				}
			})
		cohortPreviewExportService.getCohortGlobalConfig()
			.then(({ data }) => {
				if (data.code === 200) {
					setEnableCohortOps(data?.response?.ENABLE_COHORT_OPS)
				} else {
					console.error(data?.response, 'no data')
				}
			})
	};
 
	const onChangeTreeData = (name, value) => {
		setTreeData({ ...treeData, [name]: value });
	}
 
	useEffect(() => {
		retrieveConnectionsList();
	}, [])
 
	return (
		<div className="row">
			<div className="d-flex col-md-6">
				<FormControl variant="outlined" size="small" className="flex-grow-1 mr-2 mb-4">
					<InputLabel id="demo-simple-select-outlined-label">Choose Connection</InputLabel>
					<Select
						label="Choose Connection"
						labelId="demo-simple-select-outlined-label"
						id="demo-simple-select-outlined"
						value={treeData?.connectionName}
						onChange={e => onChangeTreeData(`connectionName`, e.target.value)}
					>
						<MenuItem value=""><em>None</em></MenuItem>
						{connectionsList.map((item, i) => <MenuItem key={i} value={item}>{item}</MenuItem>)}
					</Select>
				</FormControl>
				<Popover arrow placement="rightTop" overlayStyle={{ maxWidth: "500px", width: "100%" }}
					content={<img className="img-fluid p-1" src={`${URL}data-configurations-settings.png`} alt="Example Analysis" />}>
					<Icon fontSize="medium" className="text-primary m-1 mt-2">info_outlined</Icon>
				</Popover>
			</div>
			{enableCohortOps && <div className="col-md-6">
				<FormControlLabel
					control={
					<Switch
						checked={treeData?.enableCohortOps}
						onChange={e => onChangeTreeData(`enableCohortOps`, e.target.checked)}
						name="enableCohortOps"
						color="primary"
					/>
					}
					label="CohortOps - Instant View"
				/>
			</div>}
			<div className="col-md-12">
			<h6 className="mb-1">Global Query</h6>
			<FormControl variant="outlined" className="w-100 rounded p-1" style={isFocused ? {border: "2px solid #007bff"} : {border: "1px solid #ccc"}}>
				<SQLEditor
					className="w-100 border-0" height="400px"
					value={treeData?.globalPrimaryQuery}
					callChange={value => onChangeTreeData(`globalPrimaryQuery`, value)}
					onChange={() => {}}
					dataNodeIndex={0}
					onFocus={() => setFocused(true)}
					onBlur={() => setFocused(false)}
					name="UNIQUE_ID_OF_DIV"
					editorProps={{ $blockScrolling: true }}
				/>
			</FormControl>
				<FormHelperText className='helperTextGlobelQuery' id="data-info-text">
                    <em>Note: Time and uid columns are required to generate the trendline and cohort data. The Global Query should not end with semicolon( ; )</em>
                </FormHelperText>
			</div>
		</div>
	);
}
 
export default DataConfigurations;
