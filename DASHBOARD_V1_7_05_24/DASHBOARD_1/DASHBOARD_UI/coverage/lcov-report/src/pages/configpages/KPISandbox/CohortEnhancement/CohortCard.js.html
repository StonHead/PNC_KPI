import { Icon } from '@mui/material';
import { URL } from '../../../../config';
import { useState } from 'react'
 
const CohortCard = ({ x, onEditMode, openPublishedCohort, setDeleteAnalysisModalHandler }) => {
    const [hover, setHover] = useState(false)
 
    const getCohortCardStatus = (status) => {
        if (status === 'DRAFT') {
            return <p className="text-center text-white position-absolute rounded fs11 mt-1 px-1 mr-2" style={{ backgroundColor: '#DE350B' }}>DRAFT</p>
        } else if (status === 'PUBLISHED_ERROR') {
            return <p className="text-center text-white position-absolute rounded fs11 mt-1 px-1 mr-2" style={{ backgroundColor: '#DE350B' }}>ERROR</p>
        } else if (status === 'PUBLISHED_SUCCESS') {
            return <p className="text-center text-white position-absolute rounded fs11 mt-1 px-1 mr-2" style={{ backgroundColor: '#00875A' }}>PUBLISHED</p>
        } else if (status === 'READY_FOR_PUBLISH') {
            return <p className="text-center text-white position-absolute rounded fs11 mt-1 px-1 mr-2" style={{ backgroundColor: 'rgb(77 36 188 / 80%)', zIndex: '1' }}>READY FOR PUBLISH</p>
        } else {
            return <p className="text-center text-white position-absolute rounded fs11 mt-1 px-1 mr-2" style={{ backgroundColor: '#EDD68A', zIndex: '1' }}>IN PROGRESS</p>
        }
    }
 
    const getCohortCardImg = (status, analysisType) => {
        if (status === 'DRAFT') {
            return <img className="img-fluid cursor-pointer" src={`${URL}previewCohortCard.png`} alt="Preview Cohort Card" />
        } else if (status === "PUBLISHED_SUCCESS") {
            if (analysisType === 'model generated cluster') {
                return <img className="img-fluid cursor-pointer" src={`${URL}model-generated-cluster.svg`} alt="Model Generated Cohort Card" />
            } else {
                return <img className="img-fluid" src={`${URL}CSMCohortCard.png`} alt="CSM Cohort Card" />
            }
        } else if (status === "PUBLISHED_ERROR") {
            return (
                <>
                    <img className="img-fluid cursor-pointer" src={`${URL}cohortCardError.png`} alt="Error Cohort Card" />
                    <div className="row mt-4 bold500 position-absolute" >
                        <Icon className="col-2 ml-3 mt-1 p-0 text-right" style={{ color: '#EDD68A' }}>error</Icon>
                        <p className="col-7 m-0 fs12" style={{ color: '#EDD68A' }}>An error was occurred Rerun your analysis</p>
                    </div>
                </>)
        }
        else if (status === 'READY_FOR_PUBLISH') {
            return (
                <>
                    <img className="img-fluid" src={`${URL}previewCohortCard.png`} alt="Preview Cohort Card" />
                    <div className="analysis-in-progress">
                        <div className="mt-4">
                            <p className="text-white fs14">Queued...</p>
                        </div>
                    </div>
                </>
            )
        } else {
            return (
                <>
                    <img className="img-fluid" src={`${URL}previewCohortCard.png`} alt="Preview Cohort Card" />
                    <div className="analysis-in-progress">
                        <div className="mt-4">
                            <p className="text-white fs14">Generating...</p>
                        </div>
                    </div>
                </>
            )
        }
    }
 
    let editableDeletable = x.status === "PUBLISHED_SUCCESS" || x.status === "DRAFT" || x.status === "PUBLISHED_ERROR";
 
    return (
        <div className="mb35 px-0"
            onMouseEnter={() => setHover(true)}
            onMouseLeave={() => setHover(false)}>
            <div className="cohort-card" >
                <div className="d-flex justify-content-end position-relative">
                    <div className={`cohort-card-hover ${!hover && 'd-none'}`}>
                        {editableDeletable && <Icon data-testid='edit-btn' className="fs24 text-white mx-2 cursor-pointer" onClick={() => onEditMode(x.cohortMasterId, x.analysisId)}>edit</Icon>}
                        {x.status === "PUBLISHED_SUCCESS" && <Icon data-testid='view-btn' className="fs24 text-white mx-2 cursor-pointer" onClick={() => openPublishedCohort(x.cohortMasterId, x.analysisId)}>pageview</Icon>}
                        {editableDeletable && <Icon data-testid='delete-btn' className="fs24 text-white mx-2 cursor-pointer" onClick={() => setDeleteAnalysisModalHandler(x.cohortMasterId, x.analysisId)}>delete</Icon>}
                    </div>
                    {getCohortCardImg(x.status, x.analysisType)}
                    {getCohortCardStatus(x.status)}
                    <div className="row g3 text-white fs11 rounded position-absolute justify-content-around align-items-center px-1 m-0" style={{ backgroundColor: '#5E6C84', left: '5px', bottom: '5px' }}>
                        <Icon className='fs13'>person</Icon>
                        <p className="m-0">{x.createdBy}</p>
                    </div>
                </div>
 
                <div className="d-flex mx-2 mt-1 justify-content-between">
                    <p className="fs14 bold500 mb-0">{x.analysisName}</p>
                    <div className="text-right text-muted"><Icon className='fs14'>group</Icon></div>
                </div>
                <div className="card-description">
                    <p className="text-overflow-ellipsis-3 ">{x.description}</p>
                </div>
            </div>
        </div>
    )
}
 
export default CohortCard
