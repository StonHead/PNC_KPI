import React, { useEffect, useState } from "react";
import { TabContext, TabList, TabPanel } from "@mui/lab";
import { Box, Button, Tab } from "@mui/material";
import AnalystReview from "./AnalystReview";
import Settings from "./CCSettings/Settings";
import { Modal } from "antd";
import MajorDisrupters from "./Disrupters/MajorDisrupters";
import Highlights from "./Highlights/Highlights";
import SelectInsights from "./SelectInsights/SelectInsights";
import { useSnackBar } from "../../../../hooks";
import { createControlCenter, getControlCenterInsights, getControlCenterKeyTakeaway, getControlCenterMajorDisruptor, getKpiTreesForGroup } from "../../../../services/controlCenterService";
import { insightsColors } from "../../../../utilities/AllTables";
import _ from "lodash";
 
const emptyData = {
  name: "",
  groupName: "",
  objective: "",
  keyTakeaway: "",
  selectedInsights: [],
  analystNotes: "",
}
 
const Dashboard = ({
  open,
  toggleModal,
  setToggleModal,
  groupActive,
  getControlCenterCardsHandler
}) => {
  const { snackBar } = useSnackBar();
 
  const [currentStep, setCurrentStep] = useState(1);
  const [visitedTabs, setVisitedTabs] = useState([1]);
 
  const tabsList = ['Objective', 'Primary Insights', 'Primary Highlights', 'Secondary Insights', 'Secondary Hightlights', 'Major Disruptors', 'Key Takeaway'];
 
  const tabTitles = [
    "Objective",
    "Selection of Primary Insights",
    "Ranking of Primary Insights",
    "Selection of Secondary Insights",
    "Ranking of Secondary Insights",
    "Major Disruptors",
    "Key Takeaway",
  ];
 
  const [selectedRowKeysPrimary, setSelectedRowKeysPrimary] = useState([]);
  const [selectedRowKeysSecondary, setSelectedRowKeysSecondary] = useState([]);
  const [selectedDataPrimary, setSelectedDataPrimary] = useState([]);
  const [selectedDataSecondary, setSelectedDataSecondary] = useState([]);
  const [objective, setObjective] = useState("");
  const [controlCenterData, setControlCenterData] = useState(emptyData);
  // bookmarkedInsights
  const [bookmarkedInsightsPrimary, setBookmarkedInsightsPrimary] = useState([]);
  const [bookmarkedInsightsSecondary, setBookmarkedInsightsSecondary] = useState([]);
  // dataSourceInsights
  const [dataSourcePrimary, setDataSourcePrimary] = useState([]);
  const [dataSourceSecondary, setDataSourceSecondary] = useState([]);
  // Diruptors
  const [dataDisrupters, setDataDisrupters] = useState([]);
 
  const handleControlCenterData = (name, value) => {
    setControlCenterData(prev => {
      return { ...prev, [name]: value }
    });
  }
 
  const handleClose = () => {
    setToggleModal(!toggleModal);
    setControlCenterData(emptyData);
    setSelectedRowKeysPrimary([]);
    setBookmarkedInsightsPrimary([]);
    setSelectedRowKeysSecondary([]);
    setBookmarkedInsightsSecondary([]);
    setDataSourcePrimary([]);
    setDataSourceSecondary([]);
    setCurrentStep(1);
    setVisitedTabs([1]);
  }
 
  const publishHandler = () => {
    const payLoadData = {
      ..._.omit(controlCenterData,['analystNotes']),
      selectedInsights: _.map(controlCenterData.selectedInsights, object => {
        return _.omit(object, ['key', 'isPrimary'])
    })
    };
    createControlCenter(payLoadData)
      .then(({ data }) => {
        if (data.code === 201) {
          snackBar({
            status: true,
            severity: "success",
            message: `${data.response}!`,
          });
          handleClose();
          getControlCenterCardsHandler(groupActive)
        }
      })
      .catch(e => {
        snackBar({
          status: true,
          severity: "error",
          message: e.response.data.message,
        });
        handleClose();
      })
  };
 
  //   tab handlers start
  const handleChange = (event, newValue) => {
    if (currentStep < newValue) {
      if (!validateSettingsNext()) return false;
    }
    setCurrentStep(newValue);
    setVisitedTabs((prevVisitedTabs) => [...prevVisitedTabs, newValue]);
  };
  const handleNextTab = () => {
    handleChange(null, currentStep + 1);
  };
 
  const handleBackTab = () => {
    handleChange(null, currentStep - 1);
  };
  //   tab handlers end
 
  const validateSettingsNext = () => {
    if (!controlCenterData.name.trim()) {
      snackBar({ status: true, severity: "error", message: "Please Enter Control Center Name!" });
      return false;
    }
    if (!controlCenterData.objective.trim()) {
      snackBar({ status: true, severity: "error", message: "Please Enter Objective!" });
      return false;
    }
    if (currentStep == 2 && !selectedRowKeysPrimary.length) {
      snackBar({ status: true, severity: "error", message: "Please Select an Insight!" });
      return false;
    }
    // if (currentStep == 4 && !selectedRowKeysSecondary.length) {
    //   snackBar({ status: true, severity: "error", message: "Please Select an Insight!" });
    //   return false;
    // }
    return true;
  };
 
  // Insights API
  const getBookmarkedInsights = () => {
    const payload = {
      bookmarked: true,
      group: groupActive,
    }
    getControlCenterInsights({ ...payload, primary: true })
      .then(({ data }) => {
        if (data.code === 200) {
          setBookmarkedInsightsPrimary(data.response)
          setDataSource(data.response, setDataSourcePrimary);
        }
      })
      .catch((error) => {
        console.error("Error in Primary Insights:", error);
      });
    getControlCenterInsights({ ...payload, primary: false })
      .then(({ data }) => {
        if (data.code === 200) {
          setBookmarkedInsightsSecondary(data.response)
          setDataSource(data.response, setDataSourceSecondary);
        }
      })
      .catch((error) => {
        console.error("Error in Secondary Insights:", error);
      });
  };
 
  //Major Disruptor API 
  const getMajorDisruptersData = () => {
    getControlCenterMajorDisruptor({ group: groupActive }).then(
      ({ data }) => {
        if (data.code === 200) {
          setDataDisrupters(data.response);
        }
      },
      (error) => {
        console.log(error);
      }
    );
  };
 
  // KeyTakeaway
  const getAnalystReview = () => {
    getControlCenterKeyTakeaway({ group: groupActive })
      .then(({ data }) => {
        if (data.code === 200) {
          handleControlCenterData("keyTakeaway", data.response);
        }
      })
      .catch((error) => {
        console.log(error);
      })
  };
 
  const setDataSource = (data, setSource) => {
    const dataSource = data?.map((item, idx) => {
      return { ...item, id: `${item.id}`, key: idx, bgColor: insightsColors.hasOwnProperty(item.type) ? insightsColors[item.type] : "#b6b6b6" }
    })
    setSource(dataSource);
  }
 
  const updateOrderInsightValues = (data) => {
    const primaryItems = data.filter(item => item.isPrimary);
    const nonPrimaryItems = data.filter(item => !item.isPrimary);
 
    const setOrder = (items, startOrder) => {
        return items.map((item, index) => ({
            ...item,
            order: startOrder + index + 1,
        }));
    };
 
    const result = [
        ...setOrder(primaryItems, 0),
        ...setOrder(nonPrimaryItems, 0),
    ];
 
    return result;
  }
 
  const getAllData = () => {
    handleControlCenterData("groupName", groupActive);
    getBookmarkedInsights();
    getMajorDisruptersData();
    getAnalystReview();
  }
 
  useEffect(() => {
    open && getAllData()
  }, [open])
  // useEffect(() => {
  //   groupActive !== "" && getKpiTreesForGroupHandler(groupActive)
  // }, [groupActive])
 
  return (
    <Modal
      open={open}
      width={1300}
      style={{ top: 40 }}
      onCancel={handleClose}
      className={"create-modification"}
      closable={true}
      footer={[
        <div className="bg-white d-flex flex-column">
          <div className="bg-white pb-4 d-flex justify-content-center">
            {currentStep > 1 && (
              <Button
                className="text-primary-blue text-capitalize mx-2"
                onClick={handleBackTab}
              >
                Back
              </Button>
            )}
            {currentStep < 7 ? (
              <Button
                className="text-white bg-primary-blue text-capitalize"
                onClick={handleNextTab}
              // disabled={(currentStep==2 && !selectedRowKeysPrimary.length) || (currentStep==4 && !selectedRowKeysSecondary.length)}
              >
                Next
              </Button>
            ) : (
              <Button
                className="text-white bg-primary-blue text-capitalize"
                onClick={publishHandler}
              >
                Publish
              </Button>
            )}
          </div>
          <TabContext value={currentStep}>
            <TabList
              onChange={handleChange}
              className="br8 tab-theme-1 bg-grey border-grey"
              textColor="primary"
              indicatorColor="primary"
              centered
              variant="fullWidth"
            >
              <Tab disabled={!visitedTabs.includes(1)} className={`no-outline text-capitalize ${currentStep === 1 && 'bg-white'}`} value={1} icon={<><span className={`text-width px-2 ${currentStep === 1 && 'bg-white'}`}>{tabsList[0]}</span></>} iconposition="start" />
              <Tab disabled={!visitedTabs.includes(2)} className={`no-outline text-capitalize ${currentStep === 2 && 'bg-white'}`} value={2} icon={<><span className={`text-width px-2 ${currentStep === 2 && 'bg-white'}`}>{tabsList[1]}</span></>} iconposition="start" />
              <Tab disabled={!visitedTabs.includes(3)} className={`no-outline text-capitalize ${currentStep === 3 && 'bg-white'}`} value={3} icon={<><span className={`text-width px-2 ${currentStep === 3 && 'bg-white'}`}>{tabsList[2]}</span></>} iconposition="start" />
              <Tab disabled={!visitedTabs.includes(4)} className={`no-outline text-capitalize ${currentStep === 4 && 'bg-white'}`} value={4} icon={<><span className={`text-width px-2 ${currentStep === 4 && 'bg-white'}`}>{tabsList[3]}</span></>} iconposition="start" />
              <Tab disabled={!visitedTabs.includes(5)} className={`no-outline text-capitalize ${currentStep === 5 && 'bg-white'}`} value={5} icon={<><span className={`text-width px-2 ${currentStep === 5 && 'bg-white'}`}>{tabsList[4]}</span></>} iconposition="start" />
              <Tab disabled={!visitedTabs.includes(6)} className={`no-outline text-capitalize ${currentStep === 6 && 'bg-white'}`} value={6} icon={<><span className={`text-width px-2 ${currentStep === 6 && 'bg-white'}`}>{tabsList[5]}</span></>} iconposition="start" />
              <Tab disabled={!visitedTabs.includes(7)} className={`no-outline text-capitalize ${currentStep === 7 && 'bg-white'}`} value={7} icon={<><span className={`text-width px-2 ${currentStep === 7 && 'bg-white'}`}>{tabsList[6]}</span></>} iconposition="start" />
            </TabList>
          </TabContext>
        </div>,
      ]}
      title={
        <div className="d-flex">
          <div className="d-flex title-name">
            <div className="font-weight-bold text-primary-blue title-row mb-1">
              Configure Control Center
            </div>
            <div className="bar-line mx-4"></div>
            <div className="text-primary-blue title-row mb-1">
              {groupActive}
            </div>
            <div className="bar-line mx-4"></div>
            <div className="sub-title-row">{tabTitles[currentStep - 1]}</div>
          </div>
        </div>
      }
    >
      <Box className="w-100">
        <div>
          <TabContext value={currentStep}>
            <TabPanel value={1}>
              <Settings
                objective={objective}
                setObjective={setObjective}
                groupActive={groupActive}
                controlCenterData={controlCenterData}
                setControlCenterData={handleControlCenterData}
              />
            </TabPanel>
            <TabPanel value={2}>
              <SelectInsights
                isPrimary={true}
                tags={true}
                selectedRowKeys={selectedRowKeysPrimary}
                setSelectedRowKeys={setSelectedRowKeysPrimary}
                selectedData={selectedDataPrimary}
                setSelectedData={setSelectedDataPrimary}
                controlCenterData={controlCenterData}
                setControlCenterData={setControlCenterData}
                bookmarkedInsights={bookmarkedInsightsPrimary}
                dataSource={dataSourcePrimary}
                updateOrderInsightValues={updateOrderInsightValues}
              />
            </TabPanel>
            <TabPanel value={3}>
              <Highlights
                tags={true}
                setSelectedRowKeys={setSelectedRowKeysPrimary}
                selectedData={selectedDataPrimary}
                setSelectedData={setSelectedDataPrimary}
                controlCenterData={controlCenterData}
                setControlCenterData={setControlCenterData}
                setDataSource={setDataSourcePrimary}
                dataSource={dataSourcePrimary}
              />
            </TabPanel>
            <TabPanel value={4}>
              <SelectInsights
                isPrimary={false}
                selectedRowKeys={selectedRowKeysSecondary}
                setSelectedRowKeys={setSelectedRowKeysSecondary}
                selectedData={selectedDataSecondary}
                setSelectedData={setSelectedDataSecondary}
                controlCenterData={controlCenterData}
                setControlCenterData={setControlCenterData}
                bookmarkedInsights={bookmarkedInsightsSecondary}
                dataSource={dataSourceSecondary}
                updateOrderInsightValues={updateOrderInsightValues}
              />
            </TabPanel>
            <TabPanel value={5}>
              <Highlights
                setSelectedRowKeys={setSelectedRowKeysPrimary}
                selectedData={selectedDataSecondary}
                setSelectedData={setSelectedDataSecondary}
                controlCenterData={controlCenterData}
                setControlCenterData={setControlCenterData}
                setDataSource={setDataSourceSecondary}
                dataSource={dataSourceSecondary}
              />
            </TabPanel>
            <TabPanel value={6}>
              <MajorDisrupters
                controlCenterData={controlCenterData}
                setControlCenterData={handleControlCenterData}
                setDataDisrupters={setDataDisrupters}
                dataDisrupters={dataDisrupters}
              />
            </TabPanel>
            <TabPanel value={7}>
              <AnalystReview
                controlCenterData={controlCenterData}
                setControlCenterData={handleControlCenterData}
              />
            </TabPanel>
          </TabContext>
        </div>
      </Box>
    </Modal>
  );
};
 
export default Dashboard;
