import React from "react";
import { useParams, Link, useSearchParams } from "react-router-dom";
import { Button, CircularProgress } from '@mui/material';
import { Modal } from "antd";
 
import KPICalculatedTree from "./KPICalculatedTree";
import KPIVisualConfiguration from "./KPIVisualConfiguration";
import ErrorCard from "../../../components/ErrorCard";
 
import { useKPI, useMenu, useTheme } from "../../../hooks";
import { CONTEXT, URL } from "../../../config";
import kpiService from "../../../services/kpiService";
 
 
const KPICalculatedConfiguration = ({ navigate }) => {
  const { treeId } = useParams();
  const { isError } = useKPI();
  const [analysisStatus, setAnalysisStatus] = React.useState('');
  const [ editMode, setEditMode ]  = React.useState(null);
 
  const getCalculatedTreeList = () => {
    kpiService
      .getCalculatedTreeList()
          .then(({data}) => {
            if(data.code === 200){
          const treeArr = (data.response).filter(item => item.kpiTreeId === parseInt(treeId));
                  if(treeArr.length){
            setEditMode(false);
                  }else{
            setEditMode(true);
          }
 
            }else{
          setEditMode(true);
        }
      })
      .catch(error => console.error(error.message))
  };
 
  React.useEffect(() => {
    if(treeId !== `new`){
      getCalculatedTreeList()
    }
	},[])
 
  if(isError.length){
    return <ErrorCard title={``} detail={isError} />;
  }
 
  if(treeId === `new` && analysisStatus !== 'New'){
    return <NewAnalysis navigateURL={navigate} setAnalysisStatus={setAnalysisStatus} editAnalysis={setEditMode} />
  };
 
  return editMode ? <KPIVisualConfiguration editAnalysis={setEditMode} analysisStatus={analysisStatus} setAnalysisStatus={setAnalysisStatus} showTree={true} /> : <KPICalculatedTree editAnalysis={setEditMode} showTree={true} navigate={navigate} />;
}
 
const NewAnalysis = props => {
  const { isDarkMode  } = useTheme();
  const [searchParams] = useSearchParams();
  const { setTreeData, setKpiData, setSampleKPITree } = useKPI();
  const [isProgress, setIsProgress] = React.useState(false);
  const { routes } = useMenu();
  // let navigate = useNavigate();
 
  const newAnalysis = analysisId => {
    const assetMenu = routes.filter(item => `assetClassHome` in item);
    const domain = assetMenu.length && assetMenu[0]['children'].filter(x => x.route === searchParams.get("domain"));
 
    let newSampleKPITree = {
      attributes: [],
      children: [],
      name: "",
      nodeId: "1",
      relation: "",
      error: "No Title",
      nodeSvgShape: {shape: 'rect'},
    }
    let newKpiData = {
      nodeId: "1",
      parentId: [],
      childId: [],
      children: [],
      dataNode: false,
      dataValue: [{ key: '', name: '', alias: '', unit: '', polarity: 'Positive', show: true, value: "" }],
      formula: [{ key: '', name: '', alias: '', unit: '', polarity: 'Positive', show: true, value: [], sameasprimary: false }],
      displayName: "",
      relation: "",
      error: "No Title",
      height: 120,
      width: 200,
      x: 250,
      y: 2
    }
    let newTreeData = {
      analysisId,
      analysisGoal: "",
      analysisName: "",
      cohorts: [],
      drivers: [''],
      kpiTreeStatus: "Saved",
      colorThreshold: "automatic",
      kpiTreeDomain: domain.length ? domain[0].title : "Others",
      groupName: domain.length ? domain[0].title : "Others",
      chosenFilter: [{ label: "", name: "", values: [''] }],
      kpiTreeParameter: [{ key: "", parameterName: "", parameterDescription: "", parameterType: "", parameterValue: "" }],
      treeDate: { createdDate: new Date().toLocaleDateString(), modifiedDate: new Date().toLocaleDateString() },
      connectionName: "",
      globalPrimaryQuery: "",
      enableCohortOps: false,
      polarity: null,
      unit: null
    }
    setTreeData(newTreeData);
    setKpiData([newKpiData]);
    setSampleKPITree([newSampleKPITree])
  }
 
  const getSequence = () => {
    setIsProgress(true);
    kpiService
      .getSequence()
      .then(({ data }) => {
        if (data.code === 200) {
          newAnalysis(data.response.nextval);
          props.setAnalysisStatus('New');
          props.editAnalysis(true);
          props.navigateURL(CONTEXT + `analysis/` + data.response.nextval, { replace: true });
        }
      })
      .finally(() => setIsProgress(false));
  }
 
  return (
    <Modal open={true} closable={false} className= { `${"KPIModal"} ${isDarkMode ? 'theme-dark' : 'theme-default'}`}
      title={<h5 className="font-weight-bold text-primary-blue mb-0">Create a New Analysis</h5>}
      style={{ minWidth: '90%', top: 50 }}
      footer={[
        <div className="d-flex justify-content-between">
          <Button size="medium" variant="outlined" className="box-shadow-none" component={Link} to={CONTEXT+'analysis-hub/'+searchParams.get("domain")}>Cancel</Button>
          <Button color="primary" size="medium" variant="contained"
            className={`box-shadow-none ${isProgress ? 'bg-light text-muted' : 'bg-primary-blue text-white'}`}
            disabled={isProgress} onClick={() => getSequence()}>
            Choose Settings {isProgress && <CircularProgress size={14} className="text-muted ml-2" />}
          </Button>
        </div>
      ]}
      >
      <div className="row">
        <div className="col-md-6 px-md-5">
          <h6 className="my-3">An Analysis can inform you about <b>Anomalies</b> and <b>Contributing Cohort</b> details for each node in a KPI Tree.</h6>
          <h6 className="mb-3">You can view how your node data changes over time to help<b> inform present and future situations</b></h6>
          <h6>First, we will name and define the goals of your Analysis, then define the settings.</h6>
        </div>
        <div className="col-md-6 px-md-5">
          <img className="img-fluid" src={`${URL}image_kpi-tree_domestic-revenue.svg`} alt="Example Analysis" />
        </div>
      </div>
    </Modal>
  );
}
 
export default KPICalculatedConfiguration;
